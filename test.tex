\input{dc.tex}

% #1: test id
% #2: input to \dc
% #3: expected result
\def\test#1#2#3{
    \dc{#2}
    \ifnum\dcresult=#3
    \else
        \errmessage{test #1: expected #3, got \the\dcresult}
    \fi
}

% test 1
\test{1}{1 2 + 3 - 4 * 5 /}{0}

% test 2
\test{2}{6 1 - 1 + 2 / 4 *}{12}

% test 3
\test{3}{2 3 * 6 / 1 + 4 -}{-2}

% test 4
\test{4}{100 10 / 5 / 3 * 100 - 5 +}{-89}

% test 5
\test{5}{5 3 /}{1}

\bye
